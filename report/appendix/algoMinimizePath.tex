\subsection{Minimize color switch in a path}
\begin{algorithm}[H]
	\caption{Minimize switches in path}
	\label{algo:minpath}
	\KwIn{$\path = \row{\e}{\len}, \colf:=$ a path and the color function}
	\KwOut{$\affect:=$ a path affectation minimizing the color switches}

	$\affect \gets [\colf(\e_i) \text{ for } i \in [1..k] ]$\;

	\For{$i \gets 2$ \KwTo $\len$}{
		$inter \gets \affect[i - 1] \cap \affect[i]$\;
		\If{$inter \neq \varnothing$}{
			$\affect[i] \gets inter$\;
		}
	}

	$\affect[k] = \affect[k].choose()$\;
	\For{$i = k - 2$ \KwDownTo $1$}{
		\uIf {$\affect[i+1] \in prov[i]$} {
			$\affect[i] = \affect[i+1]$\;
		}
		\Else {
			$\affect[i] = \affect[i].choose()$\;
		}
	}
	\Return $\affect$\;

\end{algorithm}
